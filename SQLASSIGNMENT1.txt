1) SELECT COUNT(*) AS total_employees FROM EMP;

2) SELECT COUNT(*) AS total_departments FROM DEPT;

3) SELECT * FROM EMP;

4) SELECT * FROM DEPT;

5) SELECT SUM(SAL) AS total_salary FROM EMP;

6) SELECT SUM(COMM) AS total_commission FROM EMP;

7) SELECT JOB 
   FROM EMP
   WHERE COMM > 0;

8) SELECT SYSDATE AS current_date FROM DUAL;

9) SELECT DEPTNO, COUNT(*) AS employees_per_department 
   FROM EMP
   GROUP BY DEPTNO;

10) SELECT SUM(SAL) AS total_salary 
    FROM EMP;

11) SELECT JOB, COUNT(*) AS employees_per_job 
    FROM EMP
    GROUP BY JOB;

12) SELECT ENAME, DEPTNO 
    FROM EMP
    ORDER BY DEPTNO;

13) SELECT EMPNO, ENAME, 
           EXTRACT(DAY FROM HIREDATE) AS hire_day,
           EXTRACT(MONTH FROM HIREDATE) AS hire_month,
           EXTRACT(YEAR FROM HIREDATE) AS hire_year
    FROM EMP;

14) SELECT * 
    FROM EMP 
    ORDER BY DEPTNO;

15) SELECT * 
    FROM EMP 
    ORDER BY JOB;

16) SELECT * 
    FROM EMP 
    ORDER BY SAL DESC;

17) SELECT * 
    FROM EMP 
    ORDER BY DEPTNO ASC, SAL DESC;

18) SELECT COUNT(*) 
    FROM EMP 
    WHERE LENGTH(ENAME) = 6;

19) SELECT MAX(SAL), MIN(SAL) 
    FROM EMP;

20) SELECT DEPTNO, MAX(SAL), MIN(SAL), AVG(SAL), SUM(SAL) AS total_salary 
    FROM EMP 
    GROUP BY DEPTNO;

21) SELECT ENAME, HIREDATE 
    FROM EMP 
    ORDER BY HIREDATE;

22) SELECT * 
    FROM EMP 
    WHERE HIREDATE = (SELECT MAX(HIREDATE) FROM EMP);

23) SELECT ENAME, HIREDATE 
    FROM EMP 
    WHERE HIREDATE = (SELECT MIN(HIREDATE) FROM EMP);

24) SELECT * 
    FROM EMP 
    WHERE HIREDATE = (SELECT MIN(HIREDATE) FROM EMP);

25) SELECT ENAME, DEPTNO, HIREDATE 
    FROM EMP 
    ORDER BY HIREDATE DESC, DEPTNO ASC;

26) SELECT * 
    FROM EMP 
    WHERE SAL >= (SELECT AVG(SAL) FROM EMP);

27) SELECT * 
    FROM EMP 
    WHERE SAL BETWEEN 2000 AND 4000;

28) SELECT ENAME AS highest_paid 
    FROM EMP 
    WHERE SAL = (SELECT MAX(SAL) FROM EMP);

29) SELECT ENAME AS lowest_paid 
    FROM EMP 
    WHERE SAL = (SELECT MIN(SAL) FROM EMP);

30) SELECT COUNT(HIREDATE) 
    FROM EMP 
    WHERE TO_CHAR(HIREDATE, 'MM') = '01';
